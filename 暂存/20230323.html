Future loadmore() async {
cancelToken?.cancel();
if(sLatlng==null&&keyword==null)return false;
if(isLoad==false)return false;
try{
cancelToken=CancelToken();
List data;

  if(sLatlng!=null){
    data = await Search.AMapSearch.searchAround(
      Search.LatLng(sLatlng!.latitude,sLatlng!.longitude),
      radius: 2000,
      page: sPoiPage,
      pageSize:sPoioffset,
    );
  }else{
    data = await Search.AMapSearch.searchKeyword(
        keyword!,
      page: sPoiPage,
      pageSize:sPoioffset,
      cityLimit: true
    );
  }
  if(data.length!=sPoioffset)isLoad=false;
  setState(() {sPoiList.addAll(data);});
  sPoiPage++;
  return true;
}catch(e){
  cancelToken=null;
  print(e);
  return false;
}

}
